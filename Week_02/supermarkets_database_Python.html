cursor.execute('''CREATE TABLE IF NOT EXISTS supermarkets_table (
        type TEXT,
        id INTEGER,
        lat REAL,
        lon REAL,
        brand TEXT,
        shop TEXT,
        city TEXT,
        street TEXT,
        housenumber TEXT,
        postcode TEXT
    )
''')

# Confirm changes to the table
